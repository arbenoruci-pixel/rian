Test G — audit queries for new test data
Run after test saves:

-- New clients created today without an order by phone/code should be 0 for the test client.
select c.id, c.code, c.full_name, c.phone, c.created_at
from public.clients c
where c.created_at::date = current_date
  and not exists (
    select 1
    from public.orders o
    where o.client_id = c.id
       or normalize_phone(o.client_phone) = normalize_phone(c.phone)
       or o.code = c.code
  );

-- Final orders should not be present as draft JSON for the same local_oid.
-- Check with the app list plus Storage drafts/<local_oid>.json if needed.

-- Expired blank reservations should be clear after the app sweep or existing RPC expiry cleanup.
select code, status, reserved_by, reserved_at, lease_expires_at
from public.base_code_pool
where status = 'reserved'
  and lease_expires_at < now();
